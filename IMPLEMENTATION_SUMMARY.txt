USAGE EXAMPLES

OPTION 1: OpenAI (Your Paid Plan)
  export AI_PROVIDER=openai
  export OPENAI_API_KEY=sk-your-api-key
  python anything_to_spec.py --programs_directory ./tests/cobol/test1/ \
    --spec_path ./outputs/spec1.yml

OPTION 2: GitHub Copilot
  export AI_PROVIDER=copilot
  python anything_to_spec.py --programs_directory ./tests/cobol/test1/ \
    --spec_path ./outputs/spec1.yml

OPTION 3: GXEAI (Legacy)
  export AI_PROVIDER=gxeai
  export SAIA_PROJECT_APITOKEN=your-token
  python anything_to_spec.py --programs_directory ./tests/cobol/test1/ \
    --spec_path ./outputs/spec1.yml

OPTION 4: Command Line Override
  python anything_to_spec.py \
    --programs_directory ./tests/cobol/test1/ \
    --spec_path ./outputs/spec1.yml \
    --provider openai
